def getCardinalDirection(REGION):
    if REGION in ['Caribe', 'Eje Cafetero - Antioquia']:
        return 'North'
    elif REGION in ['Centro Oriente', 'Llano']:
        return 'East'
    elif REGION in ['Centro Sur']:
        return 'South'
    elif REGION in ['Pacifico']:
        return 'West'
    else:
        return 'Unknown'
